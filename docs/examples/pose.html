Pose Estimation Examples

Human pose detection and analysis using YoloZone

Basic Pose Detection

from yolozone.pose import PoseDetector
import cv2

# Initialize detector
detector = PoseDetector(model="yolov8n-pose.pt")

def process_image(image_path):
    # Read image
    img = cv2.imread(image_path)
    
    # Detect poses
    results = detector.detect_pose(img, conf=0.3)
    
    # Draw poses with keypoints and skeleton
    img, poses = detector.draw_poses(img, results,
                                   draw_keypoints=True,
                                   draw_skeleton=True)
    
    # Display results
    cv2.imshow("Pose Detection", img)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

Real-time Pose Detection

def process_video(video_path):
    cap = cv2.VideoCapture(video_path)
    
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        
        # Detect poses
        results = detector.detect_pose(frame, conf=0.3)
        
        # Get keypoints
        keypoints = detector.get_keypoints(results)
        
        # Draw poses
        frame, poses = detector.draw_poses(frame, results)
        
        # Display keypoint information
        for person_idx, person_kpts in enumerate(keypoints):
            y_pos = 30
            cv2.putText(frame, f"Person {person_idx + 1}", 
                       (10, y_pos), cv2.FONT_HERSHEY_SIMPLEX,
                       0.8, (0, 255, 0), 2)
            
            # Display confidence for key joints
            joints = ["nose", "left_shoulder", "right_shoulder",
                     "left_hip", "right_hip"]
            for idx, joint in enumerate(joints):
                conf = person_kpts[idx][2]
                y_pos += 25
                cv2.putText(frame, f"{joint}: {conf:.2f}",
                           (10, y_pos), cv2.FONT_HERSHEY_SIMPLEX,
                           0.6, (0, 255, 0), 2)
        
        cv2.imshow("Pose Detection", frame)
        
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
    
    cap.release()
    cv2.destroyAllWindows()

Keypoint Analysis

# Analyze specific keypoints
def analyze_pose(keypoints):
    # Check if person is standing
    left_ankle = keypoints[15]
    right_ankle = keypoints[16]
    left_hip = keypoints[11]
    right_hip = keypoints[12]
    
    if all(kp[2] > 0.5 for kp in [left_ankle, right_ankle,
                                 left_hip, right_hip]):
        # Calculate vertical alignment
        hip_y = (left_hip[1] + right_hip[1]) / 2
        ankle_y = (left_ankle[1] + right_ankle[1]) / 2
        
        if abs(hip_y - ankle_y) > 100:
            return "Standing"
        else:
            return "Sitting/Crouching"
    
    return "Uncertain"

Analysis Tips

  • Filter by confidence scores
  • Consider missing keypoints
  • Handle occlusions
  • Use relative positions

Best Practices

  • Use appropriate model size for your needs:
    • yolov8n-pose.pt: Fastest but less accurate
    • yolov8s-pose.pt: Balanced speed/accuracy
    • yolov8l-pose.pt: More accurate but slower
  • Ensure good lighting conditions
  • Position camera for clear view of subjects
  • Handle keypoint confidence thresholds
  • Consider frame resolution vs performance

Troubleshooting

Common Issues

  • Poor Detection
    • Improve lighting
    • Adjust camera position
    • Use larger model
  • Missing Keypoints
    • Lower confidence threshold
    • Handle occlusions in code
  • Performance Issues
    • Reduce resolution
    • Use smaller model
    • Process fewer frames
